1012 /* Read a character from a stream. */
1013 static inline int jas_stream_getc2(jas_stream_t *stream)
1014 {
1015  if (--stream->cnt_ < 0)
1016  return jas_stream_fillbuf(stream, 1);
1017 
1018  ++stream->rwcnt_;
1019  return (int)(*stream->ptr_++);
1020 }

1022 static inline int jas_stream_getc_macro(jas_stream_t *stream)
1023 {
1024  if (stream->flags_ & (JAS_STREAM_ERR | JAS_STREAM_EOF | JAS_STREAM_RWLIMIT))
1025  return EOF;
1026 
1027  if (stream->rwlimit_ >= 0 && stream->rwcnt_ >= stream->rwlimit_) {
1028  stream->flags_ |= JAS_STREAM_RWLIMIT;
1029  return EOF;
1030  }
1031 
1032  return jas_stream_getc2(stream);
1033 }
1034 
1035 /* Write a character to a stream. */
1036 static inline int jas_stream_putc2(jas_stream_t *stream, jas_uchar c)
1037 {
1038  stream->bufmode_ |= JAS_STREAM_WRBUF;
1039 
1040  if (--stream->cnt_ < 0)
1041  return jas_stream_flushbuf(stream, c);
1042  else {
1043  ++stream->rwcnt_;
1044  return (int)(*stream->ptr_++ = c);
1045  }
1046 }
1047 
1048 static inline int jas_stream_putc_macro(jas_stream_t *stream, jas_uchar c)
1049 {
1050  if (stream->flags_ & (JAS_STREAM_ERR | JAS_STREAM_EOF | JAS_STREAM_RWLIMIT))
1051  return EOF;
1052 
1053  if (stream->rwlimit_ >= 0 && stream->rwcnt_ >= stream->rwlimit_) {
1054  stream->flags_ |= JAS_STREAM_RWLIMIT;
1055  return EOF;
1056  }
1057 
1058  return jas_stream_putc2(stream, c);
1059 }
