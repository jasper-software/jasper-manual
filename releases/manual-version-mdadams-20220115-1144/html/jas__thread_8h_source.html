493 static inline int jas_call_once(jas_once_flag_t *flag, void (*func)(void))
494 {
495  assert(flag);
496  assert(func);
497 #if defined(JAS_THREADS_C11)
498  call_once(flag, func);
499  return 0;
500 #elif defined(JAS_THREADS_PTHREAD)
501  return pthread_once(flag, func);
502 #elif defined(JAS_THREADS_WIN32)
503  if (InterlockedCompareExchange(&flag->status, 1, 0) == 0) {
504  (func)();
505  InterlockedExchange(&flag->status, 2);
506  } else {
507  while (flag->status == 1) {
508  /* Perform a busy wait. This is ugly. */
509  jas_thread_yield();
510  }
511  }
512  return 0;
513 #endif
514 }

555 static inline
556 int jas_thread_create(jas_thread_t *thread, int (*func)(void *), void *arg)
557 {
558  assert(thread);
559  assert(func);
560 #if defined(JAS_THREADS_C11)
561  return thrd_create(thread, func, arg) == thrd_success ? 0 : -1;
562 #elif defined(JAS_THREADS_PTHREAD)
563  thread->func = func;
564  thread->arg = arg;
565  thread->result = 0;
566  return pthread_create(&thread->id, 0, thread_func_wrapper, thread);
567 #elif defined(JAS_THREADS_WIN32)
568  uintptr_t handle;
569  thread->func = func;
570  thread->arg = arg;
571  if (!(handle = _beginthreadex(0, 0, thread_func_wrapper, thread, 0, 0))) {
572  return -1;
573  }
574  thread->id = JAS_CAST(jas_thread_id_t, handle);
575  return 0;
576 #endif
577 }

582 static inline
583 int jas_thread_join(jas_thread_t *thread, int *result)
584 {
585  assert(thread);
586 #if defined(JAS_THREADS_C11)
587  return thrd_join(*thread, result) == thrd_success ? 0 : -1;
588 #elif defined(JAS_THREADS_PTHREAD)
589  void *result_buf;
590  int ret = pthread_join(thread->id, &result_buf);
591  if (!ret) {
592  jas_thread_t *other_thread = JAS_CAST(jas_thread_t *, result_buf);
593  if (result) {
594  /* A null pointer is probably a bug. */
595  assert(other_thread);
596  *result = other_thread ? other_thread->result : 0;
597  }
598  }
599  return ret;
600 #elif defined(JAS_THREADS_WIN32)
601  DWORD w;
602  DWORD code;
603  if ((w = WaitForSingleObject(thread->id, INFINITE)) != WAIT_OBJECT_0) {
604  return -1;
605  }
606  if (result) {
607  if (!GetExitCodeThread(thread->id, &code)) {
608  CloseHandle(thread->id);
609  return -1;
610  }
611  *result = JAS_CAST(int, code);
612  }
613  CloseHandle(thread->id);
614  return 0;
615 #endif
616 }
