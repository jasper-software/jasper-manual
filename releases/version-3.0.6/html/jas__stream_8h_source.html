1050 /* Read a character from a stream. */
1051 static inline int jas_stream_getc2(jas_stream_t *stream)
1052 {
1053  if (--stream->cnt_ < 0)
1054  return jas_stream_fillbuf(stream, 1);
1055 
1056  ++stream->rwcnt_;
1057  return (int)(*stream->ptr_++);
1058 }
1059 
1060 static inline int jas_stream_getc_macro(jas_stream_t *stream)
1061 {
1062  if (stream->flags_ & (JAS_STREAM_ERR | JAS_STREAM_EOF | JAS_STREAM_RWLIMIT))
1063  return EOF;
1064 
1065  if (stream->rwlimit_ >= 0 && stream->rwcnt_ >= stream->rwlimit_) {
1066  stream->flags_ |= JAS_STREAM_RWLIMIT;
1067  return EOF;
1068  }
1069 
1070  return jas_stream_getc2(stream);
1071 }
1072 
1073 /* Write a character to a stream. */
1074 static inline int jas_stream_putc2(jas_stream_t *stream, jas_uchar c)
1075 {
1076  stream->bufmode_ |= JAS_STREAM_WRBUF;
1077 
1078  if (--stream->cnt_ < 0)
1079  return jas_stream_flushbuf(stream, c);
1080  else {
1081  ++stream->rwcnt_;
1082  return (int)(*stream->ptr_++ = c);
1083  }
1084 }
1085 
1086 static inline int jas_stream_putc_macro(jas_stream_t *stream, jas_uchar c)
1087 {
1088  if (stream->flags_ & (JAS_STREAM_ERR | JAS_STREAM_EOF | JAS_STREAM_RWLIMIT))
1089  return EOF;
1090 
1091  if (stream->rwlimit_ >= 0 && stream->rwcnt_ >= stream->rwlimit_) {
1092  stream->flags_ |= JAS_STREAM_RWLIMIT;
1093  return EOF;
1094  }
1095 
1096  return jas_stream_putc2(stream, c);
1097 }
